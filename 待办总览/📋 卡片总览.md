# 📋 待办卡片总览

```dataviewjs
// ═══════════════════════════════════════════════════════════════
//  待办卡片总览 — 全库扫描 checklist，以网格卡片展示
//  点击 checkbox → 自动在原文件中标记为完成
// ═══════════════════════════════════════════════════════════════

const currentPath = dv.current().file.path;
const pages = dv.pages('""').where(p => p.file.path !== currentPath);

let tasks = [];

for (const page of pages) {
  const content = await dv.io.load(page.file.path);
  if (!content) continue;

  const lines = content.split('\n');
  let inYaml = false;
  let inCode = false;

  for (let i = 0; i < lines.length; i++) {
    const trimmed = lines[i].trimEnd();

    if (i === 0 && trimmed === '---') { inYaml = true; continue; }
    if (inYaml && trimmed === '---') { inYaml = false; continue; }
    if (inYaml) continue;
    if (trimmed.startsWith('```')) { inCode = !inCode; continue; }
    if (inCode) continue;

    const match = trimmed.match(/^(- \[([ xX]?)\] )(.+)/);
    if (match) {
      const rawText = match[3];
      const isDone = match[2].toLowerCase() === 'x';
      const cleanText = rawText.replace(/#\S+/g, '').trim();
      const tags = (rawText.match(/#(\S+)/g) || []).map(t => t.slice(1));

      tasks.push({
        text: cleanText,
        rawText,
        isDone,
        file: page.file.path,
        line: i,
        tags,
        source: page.file.link,
        created: page.file.day || null,
      });
    }
  }
}

const undone = tasks.filter(t => !t.isDone);
undone.sort((a, b) => (a.created || 0) - (b.created || 0) || a.text.localeCompare(b.text));

const MAX = 50;
const display = undone.slice(0, MAX);
const remaining = undone.length - MAX;

const container = dv.el('div', '', { cls: 'todo-grid-container' });

if (undone.length === 0) {
  container.innerHTML = '<div class="todo-empty">🎉 所有任务已完成！</div>';
  return;
}

let html = '<div class="todo-grid">';

for (const t of display) {
  const tagsHtml = t.tags.map(tag => `<span class="todo-tag" data-tag="${tag}">#${tag}</span>`).join('');
  const dateStr = t.created
    ? `${t.created.month + 1}/${t.created.day}`
    : '';
  const srcDisplay = t.source
    ? (t.source.fileName || (typeof t.source === 'string' ? t.source.split('/').pop() : ''))
    : '';

  html += `
    <div class="todo-card" data-file="${t.file}" data-line="${t.line}">
      <input type="checkbox" class="todo-checkbox" id="cb-${t.file}-${t.line}">
      <div class="todo-body">
        ${dateStr ? `<div class="todo-date">${dateStr}</div>` : ''}
        <div class="todo-text">${t.text}</div>
        <div class="todo-meta">
          ${tagsHtml}
          <span class="todo-source">${srcDisplay}</span>
        </div>
      </div>
    </div>`;
}

html += '</div>';

if (remaining > 0) {
  html += `<div class="todo-show-more">还有 ${remaining} 个任务未显示</div>`;
}

container.innerHTML = html;

// ─── 绑定点击事件 ───
const app = dv.app || this.app;
for (const cb of container.querySelectorAll('.todo-checkbox')) {
  cb.addEventListener('change', async function () {
    const card = this.closest('.todo-card');
    const filePath = card.dataset.file;
    const lineNum = parseInt(card.dataset.line);

    const file = app.vault.getAbstractFileByPath(filePath);
    if (!file) return;

    const content = await app.vault.read(file);
    const lines = content.split('\n');
    const target = lines[lineNum];
    if (target && target.includes('- [ ]')) {
      lines[lineNum] = target.replace('- [ ]', '- [x]');
      await app.vault.modify(file, lines.join('\n'));
    }

    card.style.transition = 'all 0.3s ease';
    card.style.opacity = '0';
    card.style.transform = 'scale(0.85)';
    setTimeout(() => {
      card.remove();
      if (container.querySelectorAll('.todo-card').length === 0) {
        container.innerHTML = '<div class="todo-empty">🎉 所有任务已完成！</div>';
      }
    }, 350);
  });
}
```

> 点击 checkbox → 自动在原文件中标记完成。最多显示 50 条，按日期排序。

## ⚡ 快速说明

| 操作 | 效果 |
|------|------|
| ✅ 勾选 checkbox | 原文件 `- [ ]` → `- [x]`，卡片自动消失 |
| 📁 点击来源 | 来源文件名显示在卡片底部 |
| 🏷️ 标签 | 自动识别 `#紧急` `#重要` `#一般` 等标签 |
| 🔄 刷新 | 关闭再打开此页面，或重载 Obsidian 即可刷新 |
