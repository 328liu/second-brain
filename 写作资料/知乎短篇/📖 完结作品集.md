# 📖 知乎短篇 · 完结作品集

```dataviewjs
const folder = '"写作资料/知乎短篇/完结"';
const files = dv.pages(folder);

const coverImgFile = app.vault.getFileByPath("自媒体搞钱/图片/52628b4b5b0bc6c.jpg");
const coverImgUrl = coverImgFile ? app.vault.getResourcePath(coverImgFile) : "";

const genreKeywords = ["追妻","大女主","重生","脑洞","爽文","末日","现言","破镜重圆"];
function parseGenre(filename) {
  for (const g of genreKeywords) {
    if (filename.includes(g)) return g;
  }
  return "其他";
}

function parseWordCount(filename) {
  const m = filename.match(/(\d+\.?\d*)\s*[Ww]/);
  if (m) return m[1] + "W";
  const m2 = filename.match(/[+_-\s](\d+\.?\d*)\s*[Ww万]/);
  if (m2) return m2[1] + "W";
  return "";
}

function extractTitle(content) {
  const lines = content.split("\n").filter(l => l.trim());
  if (!lines.length) return "未命名";
  let raw = lines[0].replace(/^#+\s*/, "").trim();
  if (raw.length > 15) raw = raw.slice(0, 15) + "…";
  return raw || "未命名";
}

const tagColors = {
  "追妻": "#f48fb1",
  "大女主": "#ce93d8",
  "重生": "#90caf9",
  "脑洞": "#ffb74d",
  "爽文": "#81c784",
  "末日": "#90a4ae",
  "现言": "#81d4fa",
  "破镜重圆": "#ffd54f",
};
function tagClr(g) { return tagColors[g] || "#bdbdbd"; }

const cardData = [];
for (const f of files) {
  const content = await dv.io.load(f.file.path);
  const filename = f.file.name;
  const genre = parseGenre(filename);
  const wordCount = parseWordCount(filename);
  const title = extractTitle(content);
  cardData.push({ file: f, title, genre, wordCount });
}

cardData.sort((a, b) => a.title.localeCompare(b.title, "zh"));

dv.container.innerHTML = `
<style>

  display: grid;
  grid-template-columns: repeat(auto-fill, minmax(240px, 1fr));
  gap: 20px;
  padding: 8px 0;
}
.novel-card {
  border-radius: 14px;
  overflow: hidden;
  background: #fff;

}

  justify-content: center;

  background-position: center;
}

  position: relative;
  z-index: 1;
  font-size: 20px;
  font-weight: 700;
  color: #4a3b32;
  line-height: 1.5;

  display: -webkit-box;
  -webkit-line-clamp: 4;

}

}

}
</style>
<div class="novel-grid">
${cardData.map(c => `
<div class="novel-card" onclick="app.workspace.openLinkText('${c.file.file.name}', '${c.file.file.path}', false)">

  </div>
  <div class="novel-card-body">

      <span class="tag" style="background: ${tagClr(c.genre)}">${c.genre}</span>
    </div>
  </div>
</div>
`).join("")}
</div>
`;
```
